public static int[] TwoSum(int[] nums, int target) 
        {
            for(var i=0;i<nums.Length;i++){
                for(var j=nums.Length-1;j>i;j--){
                    if(i!=j){
                        if(nums[i]+nums[j]==target){
                            return new int[]{i,j};
                        }
                    }
                }
            }
            return new int[]{};
        }
